! --- Network Filters ---

||linkedin.com/li/track^$script
||linkedin.com/sensorCollect^$xhr
||linkedin.com/biz^$script
||linkedin.com/px^$image

! --- Cosmetic Filters ---

! Block ads and promotional content
linkedin.com##iframe[title="Advertisement"]

! News box
linkedin.com##aside[aria-label="LinkedIn News"] > *:style(display: none !important;)

! Puzzle game
linkedin.com##div:has(> div:has(> p:has-text(puzzle game)))

! Block promoted posts
linkedin.com##div.feed-shared-update-v2:has(div.update-components-header:has-text(/Promoted/))

! Block AI-generated content and suggestions
linkedin.com##div.feed-shared-update-v2:has(div.update-components-header:has-text(/Suggested/))

! Block social engagement noise (comments)
linkedin.com##:not(:matches-path(/recent-activity)) div.feed-shared-update-v2:has(div.update-components-header:has-text(/commented on this/))

! Block social engagement noise (reposts)
linkedin.com##:not(:matches-path(/recent-activity)) div.feed-shared-update-v2:has(div.update-components-header:has-text(/reposted this/))

! Block social engagement noise (likes, reactions)
linkedin.com##:not(:matches-path(/recent-activity)) div.feed-shared-update-v2:has(div.update-components-header:has-text(/likes this|loves this|finds this insightful|finds this funny|celebrates this|supports this/))

! Block notification cards (profile views, search appearances)
linkedin.com##article.nt-card:has-text(viewed your profile: Stay anonymous)
linkedin.com##article.nt-card:has-text(/You appeared in .* searches this week/)
